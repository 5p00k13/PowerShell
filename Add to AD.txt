#ADD Computer
Import-Module ActiveDirectory
$domain = "yourdomain.local"
$dc = "your-dc-hostname"  # Optional in most cases
$adminUser = "yourdomain\administrator"

$cred = Get-Credential -UserName $adminUser
Add-Computer -DomainName $domain -Credential $cred -Server $dc -Force -Restart




#Add user
function AddAD_User {
    Import-Module ActiveDirectory

    Write-Host "`n=== Create New AD User ==="

    $firstName = Read-Host "Enter First Name"
    $lastName = Read-Host "Enter Last Name"
    $username = Read-Host "Enter Username (sAMAccountName)"
    $domain = Read-Host "Enter Domain (e.g., yourdomain.local)"
    $ouPath = Read-Host "Enter OU Path (e.g., OU=Users,DC=yourdomain,DC=local)"
    $password = Read-Host "Enter Password" -AsSecureString

    $fullName = "$firstName $lastName"
    $upn = "$username@$domain"

    New-ADUser `
        -Name $fullName `
        -GivenName $firstName `
        -Surname $lastName `
        -SamAccountName $username `
        -UserPrincipalName $upn `
        -DisplayName $fullName `
        -Path $ouPath `
        -AccountPassword $password `
        -Enabled $true

    Write-Host " User '$fullName' created successfully.`n"
}
